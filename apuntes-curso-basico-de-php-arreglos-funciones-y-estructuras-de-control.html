              echo "<br>";
              echo "<br>";
              
              function suma_infinita(...$params) {
                  $total = 0;
                  foreach ($params as $numero) {
                      $total += $numero;
                  }
                  echo "El total es: $total <br>";
              }
              
              suma_infinita(1,2,3);
              suma_infinita(4,5,6);
              suma_infinita(1,2,3,4,5,6,7,8,9);
